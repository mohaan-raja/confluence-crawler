def uninstall_application(account_id,application_id)
	Sharding.select_shard_of(account_id) do
		FdMultitenant::Account.find(account_id).make_current
		app = Integration::InstalledApplication.find_by_application_id(application_id)
		begin
			app.destroy
		rescue Exception => e
			puts "Uninstall Failed #{e.message}"
		end
	end
end

def tester_f7(account_id)
    Sharding.select_shard_of(account_id) do
      failed_accounts = []
      begin
        account = FdMultitenant::Account.find_by_id(account_id)
        next if account.nil?
        account.make_current
        account.user_with_higesh_privilege.make_current
        # this check was added to check whether the account plan has the feature , join and split was added to reframe the nested array in to a single array
        if S360PlanFeatureConfig[:plans][account.subscription.plan_name.to_sym][:features].join(',').split(',').include?('fm_landing_page')
          privileges_to_be_added = [:fm_landing_page_view, :fm_landing_page_create_and_edit]
          FdMultitenant::Account.current.roles.each do |role|
            feeder_product_privileges = (role.feeder_product_abilities.map(&:to_sym) + privileges_to_be_added).uniq
            role.feeder_product_privileges = Role.feeder_product_privileges_mask(feeder_product_privileges, true)
            role.skip_editable_validation = true
            role.save
            puts "roles updated for this account"
          end
        end
      rescue Exception => e
        failed_accounts << [account_id, e.message]
      end
      if failed_accounts.present?
        puts "FAILED ACCOUNTS"
        failed_accounts.each { |arr| puts arr }
      end
    end
  end

#. Enable/Disable Temp Feature by using below script 

def quotav2(acc_id, action)
    Sharding.select_shard_of(acc_id) do
        account = FdMultitenant::Account.find_by_id(acc_id)
        return if account.blank?
        begin
          account.make_current
          account.user_with_higesh_privilege.make_current
          if (action.eql?('add'))
            account.try_registering_temp_feature!(:quotas_v2)
            puts "Feature is Added"
          elsif (action.eql?('remove'))
            account.remove_temp_feature_from_include_list(:quotas_v2)
            puts "Feature is removed"
          else
            puts "Action is not correct"
          end
          account.invalidate_sessions_version_from_client_cache
        rescue => e
          puts e.message
          puts "Failed for account #{account.id}"
        end
    end
  end
  quotav2(220915,'add')
