Query 
select AVG(weekyear.ncount) from (
select weekofyear(updated_at) week_no, COUNT(*) as ncount
FROM leads
where is_deleted=1 AND
YEAR (updated_at) = 2019
GROUP BY weekofyear(updated_at)) as weekyear
